\question{
        For  $m\leq k\leq n$, prove the identity
        $$\binom{n}{k}\binom{k}{m}=\binom{n}{m}
        \binom{n-m}{k-m}.$$
    }
    \begin{solution}
        Here
        \begin{align*}
            {n \choose k}{k \choose m}&=
            \frac{n!k!}{k!m!(n-k)!(k-m)!}\\
            &=\frac{n!(n-m)!}{m!(n-m)!(n-k)!(k-m)!}\\
            &=\frac{n!}{m!(n-m)!}\cdot
                \frac{(n-m)!}{(k-m)!\left((n-m)-(k-m)\right)!}\\
            &={n \choose m}{{n-m} \choose {k-m}}
        \end{align*}
    \end{solution}
